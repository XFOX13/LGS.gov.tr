function foxCoverEyes() { // Move paws UP to cover eyes document.getElementById('paw-left').classList.add('cover-left'); document.getElementById('paw-right').classList.add('cover-right'); document.getElementById('paw-right').classList.remove('peek-right'); } function togglePassword() { const pwd = document.getElementById('password'); const isHidden = pwd.type === 'password'; pwd.type = isHidden ? 'text' : 'password'; if (!isHidden) { // Show password → right paw moves DOWN to peek document.getElementById('paw-right').classList.remove('cover-right'); document.getElementById('paw-right').classList.add('peek-right'); } else { // Hide password → cover both eyes again foxCoverEyes(); } }
